export const canActivateUsrForServerDetail: CanActivateChildFn = (route, state) => {


  // default redirection
  const redirectUrl = inject(Router).createUrlTree(['/'])


  const promise : Promise<boolean|UrlTree> = new Promise( async (resolve) => {
    const isAuthenticated = await inject(AuthService).isAuthenticated() ;
    if (isAuthenticated) {
      resolve(true)
    } else {
      resolve(redirectUrl)
    }
  })

};

When injecting services into service use @Injectable() annotation in service where other service is injected. Best practice is to use annotation for all services. Make sure services are defined as provider at the correct hierachy. Alternatively use @Injectable({providedIn: 'root'}) decorator instead of using the providers property in the app module.

@Pipe({
  name: "abbreviate"
})
export class AbbreviatePipe implements PipeTransform {
  transform(value: string, ...args: Array<Record<string, string>>): any {
    let workValue = value
    args.forEach((arg => {
      Object.keys(arg).forEach(key => {
        workValue = workValue.replace(key, arg[key]);
      })
    }))
    return workValue;
  }
}
